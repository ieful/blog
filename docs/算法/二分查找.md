title: 二分查找
---



```javascript
    function binarySearch(arr, target) {
    let len = arr.length;
    let low = 0, heigh = len - 1;
    while (low <= heigh) {
        let mid = Math.floor((low + heigh)/2);
        let guess = arr[mid];
        if (guess > target) {
            heigh = mid - 1;
        } else if (guess < target) {
            low = mid + 1;
        } else if (arr[guess] === target) {
            return guess;
        }
    }
    return null;
}
```
