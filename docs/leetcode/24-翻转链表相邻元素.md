title: 翻转链表相邻元素
---

```javascript
// Given 1 -> 2 -> 3 -> 4
// Return 2 -> 1 -> 4 -> 3
function reverseNeiborLinkedList(head) {
    if (!head.next) {
        return head;
    }
    let curr = head;
    let pre = null;
    while (curr.next && curr.next.next) {
        let next = curr.next.next;
        curr.next.next = curr;
        curr = next;
    }
}
```
