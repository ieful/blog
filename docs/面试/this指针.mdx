### 求以下代码的输出结果

```js
var length = 10;
var obj1 = {
    length: 6,
    method: function () {
        console.log(this.length)
    }
}

var obj2 = {
    length: 3,
    method: function (fn) {
        fn();
        arguments[0]();
    }
}

obj2.method(obj1.method, obj2.method); // 求打印输出结果
```
